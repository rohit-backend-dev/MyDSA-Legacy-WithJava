    public int maxSubArray(int[] nums) {
        int max = Integer.MIN_VALUE; // store best answer
        int sum = 0;                 // running sum

        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];               // add current element

        }

        return max;
    }
