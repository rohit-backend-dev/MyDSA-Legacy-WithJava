class Solution {
    public int countPrimes(int n) {
        if (n < 3) return 0; // no primes less than 2

        // Only store odd numbers to reduce space by ~50%
        boolean[] isPrime = new boolean[n];
        Arrays.fill(isPrime, true);

        int count = n / 2; // start by assuming all odds are prime except even numbers

        for (int i = 3; i * i < n; i += 2) { // skip even numbers
            if (isPrime[i]) {
                // start marking from i*i, skip even multiples
                for (int j = i * i; j < n; j += 2 * i) {
                    if (isPrime[j]) {
                        isPrime[j] = false;
                        count--;
                    }
                }
            }
        }

        return count; // includes prime 2 automatically in count
    }
}
